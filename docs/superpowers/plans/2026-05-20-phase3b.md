```rust
// crates/engine/src/rate_limiter.rs
use anyhow::Result;
use std::time::Instant;

pub(crate) struct RateLimiter {
    last_request: Instant,
    min_interval_ms: u64,
}

impl RateLimiter {
    pub(crate) fn new(min_interval_ms: u64) -> Self {
        Self {
            last_request: Instant::now() - std::time::Duration::from_millis(min_interval_ms),
            min_interval_ms,
        }
    }
    pub(crate) fn check(&mut self) -> Result<()> {
        let elapsed = self.last_request.elapsed();
        let min_dur = std::time::Duration::from_millis(self.min_interval_ms);
        if elapsed < min_dur {
            anyhow::bail!("rate limit exceeded, retry in {}ms", (min_dur - elapsed).as_millis());
        }
        self.last_request = Instant::now();
        Ok(())
    }
}
```

```rust
use openloom_skills::SkillPermissions;
use anyhow::Result;

/// Check skill permissions before invoking.
/// Returns Ok(()) if the operation is allowed, Err(PermissionDenied) otherwise.
pub fn check_permissions(
    permissions: &SkillPermissions,
    skill_name: &str,
    params: &serde_json::Value,
) -> Result<()> {
    // Check filesystem read paths
    if let Some(paths) = &permissions.fs_read {
        if let Some(file_path) = params.get("path").and_then(|v| v.as_str()) {
            if !paths.iter().any(|allowed| file_path.starts_with(allowed)) {
                anyhow::bail!(
                    "Permission denied: {} may not read '{}' (allowed: {:?})",
                    skill_name, file_path, paths
                );
            }
        }
    }

    // Check filesystem write paths
    if let Some(paths) = &permissions.fs_write {
        if let Some(file_path) = params.get("path").and_then(|v| v.as_str()) {
            if !paths.iter().any(|allowed| file_path.starts_with(allowed)) {
                anyhow::bail!(
                    "Permission denied: {} may not write '{}' (allowed: {:?})",
                    skill_name, file_path, paths
                );
            }
        }
    }

    // Check network access
    if let Some(domains) = &permissions.network {
        if let Some(url) = params.get("url").and_then(|v| v.as_str()) {
            if !domains.iter().any(|allowed| url.contains(allowed)) {
                anyhow::bail!(
                    "Permission denied: {} may not access '{}' (allowed: {:?})",
                    skill_name, url, domains
                );
            }
        }
    }

    // Check subprocess execution
    if !permissions.subprocess {
        if params.get("command").is_some() {
            anyhow::bail!("Permission denied: {} may not execute subprocesses", skill_name);
        }
    }

    Ok(())
}
```

```rust
pub async fn cognition_snapshots(&self, cognition_id: i64) -> Result<Vec<openloom_memory::store::CognitionSnapshot>> {
    let conn = rusqlite::Connection::open(&self.db_path)?;
    let store = openloom_memory::store::CognitionStore::new(&conn);
    store.snapshots_for(cognition_id)
}

pub async fn rollback_cognition(&self, cognition_id: i64, version: i64) -> Result<()> {
    let conn = rusqlite::Connection::open(&self.db_path)?;
    let store = openloom_memory::store::CognitionStore::new(&conn);
    // 1. Find target snapshot
    let snapshots = store.snapshots_for(cognition_id)?;
    let target = snapshots.iter().find(|s| s.version == version)
        .ok_or_else(|| anyhow::anyhow!("version {} not found for cognition {}", version, cognition_id))?;
    // 2. Get the actual subject from the cognitions table
    let subject: String = conn.query_row(
        "SELECT subject FROM cognitions WHERE id = ?1",
        rusqlite::params![cognition_id], |row| row.get(0),
    )?;
    // 3. Restore snapshot as a new version
    store.insert(&subject, &target.trait_name, &target.value, target.confidence, target.evidence_count)?;
    // 4. Emit event
    let _ = self.event_bus.send(EngineEvent::CognitionUpdated {
        trait_name: target.trait_name.clone(), old_value: String::new(),
        new_value: target.value.clone(), confidence: target.confidence,
    });
    Ok(())
}
```

- [ ] **Step 3: 创建 CognitionAuditPanel.tsx**

`F:/openLoom/web/src/components/CognitionAuditPanel.tsx` — 类似 PersonaPanel 但增强为可查看版本历史 + 回滚：

```tsx
import React, { useState, useEffect } from 'react';

interface Trait {
    id: number; trait: string; value: string; confidence: number; evidence_count: number; version: number;
}
interface Snapshot {
    id: number; version: number; trait_name: string; value: string; snapshot_at: number;
}

export default function CognitionAuditPanel() {
    const [traits, setTraits] = useState<Trait[]>([]);
    const [selected, setSelected] = useState<Trait | null>(null);
    const [snapshots, setSnapshots] = useState<Snapshot[]>([]);
    const [loading, setLoading] = useState(true);

    const fetchTraits = async () => {
        const c: any = await window.openloom?.send('memory.cognitions', { subject: 'USER', limit: 50 });
        setTraits(c.cognitions || []);
        setLoading(false);
    };

    useEffect(() => { fetchTraits(); }, []);

    const viewSnapshots = async (trait_: Trait) => {
        setSelected(trait_);
        const s: any = await window.openloom?.send('memory.cognition_snapshots', { cognition_id: trait_.id });
        setSnapshots(s.snapshots || []);
    };

    const rollback = async (cognitionId: number, version: number) => {
        await window.openloom?.send('memory.cognition_rollback', { cognition_id: cognitionId, version });
        fetchTraits();
        setSelected(null);
    };

    return (
        <div className="p-6 overflow-y-auto">
            <h2 className="text-xl font-bold mb-4">Cognition Audit</h2>
            {loading ? <p>Loading...</p> : traits.length === 0 ? <p>No cognitions yet.</p> : (
                <div className="space-y-2">
                    {traits.map(t => (
                        <div key={t.id} className="p-3 bg-gray-800 rounded cursor-pointer hover:bg-gray-750" onClick={() => viewSnapshots(t)}>
                            <div className="flex justify-between">
                                <span>{t.trait}: {t.value} (v{t.version})</span>
                                <span className="text-gray-400">{(t.confidence * 100).toFixed(0)}%</span>
                            </div>
                        </div>
                    ))}
                </div>
            )}
            {selected && snapshots.length > 0 && (
                <div className="mt-6 p-4 bg-gray-900 rounded">
                    <h3 className="font-bold mb-2">Version History: {selected.trait}</h3>
                    {snapshots.map(s => (
                        <div key={s.id} className="flex justify-between p-2 border-b border-gray-700">
                            <span>v{s.version}: {s.value}</span>
                            <button onClick={() => rollback(selected.id, s.version)}
                                className="px-2 py-0.5 bg-yellow-600 rounded text-xs">Rollback</button>
                        </div>
                    ))}
                </div>
            )}
        </div>
    );
}
```
