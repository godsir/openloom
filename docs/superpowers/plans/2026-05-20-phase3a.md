- [ ] **Step 3: 实现 load_blocking()**

```rust
impl InferenceEngine {
    #[cfg(feature = "llama")]
    pub fn load_blocking(model_path: &std::path::Path, n_gpu_layers: usize) -> anyhow::Result<Self> {
        use llama_cpp_2::*;
        use std::num::NonZeroU32;

        if !model_path.exists() {
            tracing::warn!(path = %model_path.display(), "model file not found, inference unavailable");
            return Ok(Self { sender: None });
        }

        let backend = LlamaBackend::init()
            .map_err(|e| anyhow::anyhow!("llama backend init: {}", e))?;

        let model = LlamaModel::load_from_file(
            &backend,
            model_path,
            &LlamaModelParams::default().with_n_gpu_layers(n_gpu_layers as u32),
        ).map_err(|e| anyhow::anyhow!("model load: {}", e))?;

        let ctx_params = LlamaContextParams::default()
            .with_n_ctx(NonZeroU32::new(4096).unwrap());
        let ctx = model.new_context(&backend, ctx_params)
            .map_err(|e| anyhow::anyhow!("context create: {}", e))?;

        let (tx, rx) = mpsc::channel::<EngineCommand>();

        std::thread::Builder::new()
            .name("llama-worker".into())
            .spawn(move || {
                let sampler = LlamaSampler::chain_simple([
                    LlamaSampler::temp(0.7),
                    LlamaSampler::top_p(0.95),
                    LlamaSampler::greedy(),
                ]);
                // Note: sampler, ctx, model, backend all stay in this thread (!Send)

                for cmd in rx {
                    match cmd {
                        EngineCommand::Complete { prompt, max_tokens, temperature: _, reply } => {
                            let result = Self::do_complete(&model, &ctx, &sampler, &prompt, max_tokens);
                            let _ = reply.send(result);
                        }
                        EngineCommand::CompleteStream { prompt, max_tokens, token_tx } => {
                            Self::do_complete_stream(&model, &ctx, &sampler, &prompt, max_tokens, token_tx);
                        }
                        EngineCommand::TokenCount { text, reply } => {
                            let count = model.str_to_token(&text, llama_cpp_2::AddBos::Never)
                                .map(|tokens| tokens.len())
                                .unwrap_or(text.chars().count() / 4);
                            let _ = reply.send(count);
                        }
                    }
                }
            })?;

        tracing::info!(path = %model_path.display(), n_gpu_layers, "llama model loaded");
        Ok(Self { sender: Some(tx) })
    }

    #[cfg(not(feature = "llama"))]
    pub fn load_blocking(model_path: &std::path::Path, n_gpu_layers: usize) -> anyhow::Result<Self> {
        Ok(Self { _model_path: model_path.to_path_buf(), _n_gpu_layers: n_gpu_layers })
    }
}
```

- [ ] **Step 5: 实现 complete_stream() + token_count()**

`complete_stream()` 通过 bridge pattern 将 std::mpsc → tokio::mpsc：

```rust
#[cfg(feature = "llama")]
pub async fn complete_stream(&self, req: CompletionRequest, token_tx: tokio::sync::mpsc::Sender<String>) -> anyhow::Result<()> {
    if let Some(ref sender) = self.sender {
        let (std_tx, std_rx) = mpsc::channel::<String>();
        sender.send(EngineCommand::CompleteStream {
            prompt: req.prompt.clone(),
            max_tokens: req.max_tokens,
            token_tx: std_tx,
        }).map_err(|e| anyhow::anyhow!("worker thread: {}", e))?;
        // Bridge std::mpsc → tokio::mpsc
        for token in std_rx {
            if token_tx.send(token).await.is_err() { break; }
        }
        return Ok(());
    }
    // Fallback: send whole response as one token
    let resp = self.complete(req).await?;
    let _ = token_tx.send(resp.text).await;
    Ok(())
}

pub fn token_count(&self, text: &str) -> usize {
    #[cfg(feature = "llama")]
    if let Some(ref sender) = self.sender {
        let (reply_tx, reply_rx) = oneshot::channel();
        if sender.send(EngineCommand::TokenCount { text: text.to_string(), reply: reply_tx }).is_ok() {
            if let Ok(count) = reply_rx.blocking_recv() {
                return count;
            }
        }
    }
    // Fallback estimation
    text.chars().count() / 4
}
```

```rust
use axum::extract::{Path, Query, State};
use axum::response::sse::{Event, Sse};
use futures::stream::{self, Stream};
use openloom_engine::Engine;
use openloom_inference::CompletionRequest;
use serde::Deserialize;
use std::convert::Infallible;
use std::sync::Arc;
use tokio::sync::mpsc;

#[derive(Deserialize)]
struct SseParams {
    prompt: String,
    max_tokens: Option<usize>,
}

pub async fn sse_handler(
    Path(session_id): Path<String>,
    Query(params): Query<SseParams>,
    State(engine): State<Arc<Engine>>,
) -> Sse<impl Stream<Item = Result<Event, Infallible>>> {
    let (tx, mut rx) = mpsc::channel::<String>(64);

    let engine = engine.clone();
    tokio::spawn(async move {
        let req = CompletionRequest {
            prompt: params.prompt,
            max_tokens: params.max_tokens.unwrap_or(2048),
            temperature: 0.7,
            top_p: 1.0,
            stop: Vec::new(),
            stream: true,
        };
        let _ = engine.stream_complete(req, tx).await;
    });

    let stream = stream::unfold(rx, |mut rx| async move {
        match rx.recv().await {
            Some(token) => Some((Ok(Event::default().data(token)), rx)),
            None => None,
        }
    });

    Sse::new(stream)
}
```

```tsx
async function sendMessage() {
    if (!input.trim() || loading) return;
    const userMsg: Message = { role: 'user', content: input };
    setMessages(prev => [...prev, userMsg, { role: 'assistant', content: '' }]);
    setInput('');

                updated[updated.length - 1] = { ...last, content: last.content + e.data };
            }
            return updated;
        });
    };

    es.addEventListener('done', () => { es.close(); setLoading(false); });
    es.onerror = () => { es.close(); setLoading(false); };
}
```

```rust
#[async_trait]
impl CloudClient for AnthropicClient {
    async fn complete_stream(
        &self,
        req: CompletionRequest,
        tx: tokio::sync::mpsc::Sender<String>,
    ) -> anyhow::Result<()> {
        let body = serde_json::json!({
            "model": self.model,
            "max_tokens": req.max_tokens,
            "messages": [{"role": "user", "content": req.prompt}],
            "stream": true,
        });
        let resp = self.http
            .post("https://api.anthropic.com/v1/messages")
            .header("x-api-key", &self.api_key)
            .header("anthropic-version", "2023-06-01")
            .json(&body)
            .send()
            .await?;

        if !resp.status().is_success() {
            anyhow::bail!("Anthropic API error: {}", resp.status());
        }

        use futures::StreamExt;
        let mut stream = resp.bytes_stream();
        let mut buffer = String::new();

        while let Some(chunk) = stream.next().await {
            let chunk = chunk?;
            buffer.push_str(&String::from_utf8_lossy(&chunk));

            // Parse SSE frames: "event: content_block_delta\ndata: {...}\n\n"
            while let Some(pos) = buffer.find("\n\n") {
                let frame = buffer[..pos].to_string();
                buffer = buffer[pos + 2..].to_string();

                for line in frame.lines() {
                    if line.starts_with("data: ") {
                        let json_str = &line[6..];
                        if let Ok(val) = serde_json::from_str::<serde_json::Value>(json_str) {
                            if let Some(text) = val["delta"]["text"].as_str() {
                                if tx.send(text.to_string()).await.is_err() {
                                    return Ok(()); // client disconnected
                                }
                            }
                        }
                    }
                }
            }
        }
        Ok(())
    }
}
```
