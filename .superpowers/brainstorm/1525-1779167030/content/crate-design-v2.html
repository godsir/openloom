handle_message 内部流:
async fn handle_message(&self, msg, sid) -> Result<ChatResponse> {
    let out = self.router.classify(&msg.text).await?;
    let response = match out.skill_match {
        Some(name) => self.skills.invoke(&name, params).await?,
        None => self.inference.complete(prompt).await?,
    };
    // 后台异步: memory pipeline
    let mem = self.memory.clone();
    tokio::spawn_blocking(move || mem.process(&sid, &msg.text, &out.intent));
    // 广播 token_usage 事件
    let _ = self.event_bus.send(EngineEvent::TokenUsage { ... });
    Ok(ChatResponse { response, session_id: sid, token_usage })
}
